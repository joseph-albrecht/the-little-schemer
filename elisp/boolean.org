#+BEGIN_SRC emacs-lisp
  (defun gt (num1 num2)
    "return true if num1 > num2"
    (cond
     ((eq num1 0) nil)
     ((eq num2 0) t)
     (t (gt (1- num1) (1- num2)))))
#+END_SRC
